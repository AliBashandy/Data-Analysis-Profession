\begin{frame}{Introduction}
\protect\hypertarget{introduction}{}
This analysis based on Global Country Information Data set 2023 by
Nidula Elgiriyewithana found in kaggle with public licence Attribution
4.0 International (CC BY 4.0). This comprehensive data set provides a
wealth of information about all countries worldwide, covering a wide
range of indicators and attributes. It encompasses demographic
statistics, economic indicators, environmental factors, healthcare
metrics, education statistics, and much more. With every country
represented, this data set offers a complete global perspective on
various aspects of nations, enabling in-depth analyses and cross-country
comparisons.
\end{frame}

\begin{frame}[fragile]{1. The Question?}
\protect\hypertarget{the-question}{}
Analyze the information about all countries worldwide then categorize
them to top 10 and bottom 10. Use the information to get insight about
correlation of different indicators and attributes through
visualization.

\begin{block}{2. Prepare: getting things ready.}
\protect\hypertarget{prepare-getting-things-ready.}{}
